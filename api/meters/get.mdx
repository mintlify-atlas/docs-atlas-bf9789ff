---
title: "Get Meter"
api: "GET /v1/meters/{id}"
description: "Retrieve details of a specific meter by ID."
---

## Path Parameters

<ParamField path="id" type="string" required>
  Unique identifier of the meter.
  
  **Example:** `"meter_550e8400e29b41d4a716446655440000"`
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the meter.
</ResponseField>

<ResponseField name="name" type="string">
  Human-readable name for the meter.
</ResponseField>

<ResponseField name="tenant_id" type="string">
  Tenant ID that owns this meter.
</ResponseField>

<ResponseField name="event_name" type="string">
  Name of the event being tracked.
</ResponseField>

<ResponseField name="aggregation" type="object">
  Aggregation configuration.
  
  <Expandable title="aggregation properties">
    <ResponseField name="type" type="string">
      Aggregation method.
    </ResponseField>
    
    <ResponseField name="field" type="string">
      Property name to aggregate.
    </ResponseField>
    
    <ResponseField name="expression" type="string">
      Optional CEL expression for custom aggregation.
    </ResponseField>
    
    <ResponseField name="multiplier" type="string">
      Decimal multiplier applied to the result.
    </ResponseField>
    
    <ResponseField name="bucket_size" type="string">
      Window size for bucketed aggregations.
    </ResponseField>
    
    <ResponseField name="group_by" type="string">
      Property to group by before aggregating.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="filters" type="array">
  Array of filters applied to events before aggregation.
</ResponseField>

<ResponseField name="reset_usage" type="string">
  Determines when usage resets.
</ResponseField>

<ResponseField name="status" type="string">
  Status of the meter.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp when the meter was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp when the meter was last updated.
</ResponseField>

## Example Request

```bash cURL
curl --request GET \
  --url https://api.flexprice.io/v1/meters/meter_550e8400e29b41d4a716446655440000 \
  --header 'Authorization: Bearer <token>'
```
